44 void default_mrsg_dfs_f (char** dfs_matrix, size_t chunks, size_t workers_mrsg, int replicas)
45 {
46  int r;
47  size_t chunk;
48  size_t owner;
49 
50  if (config_mrsg.mrsg_chunk_replicas >= config_mrsg.mrsg_number_of_workers)
51  {
52  /* All workers own every chunk. */
53  for (chunk = 0; chunk < config_mrsg.mrsg_chunk_count; chunk++)
54  {
55  for (owner = 0; owner < config_mrsg.mrsg_number_of_workers; owner++)
56  {
57  chunk_owner_mrsg[chunk][owner] = 1;
58  }
59  }
60  }
61  else
62  {
63  /* Ok, it's a typical distribution. */
64  for (chunk = 0; chunk < config_mrsg.mrsg_chunk_count; chunk++)
65  {
66  for (r = 0; r < config_mrsg.mrsg_chunk_replicas; r++)
67  {
68  owner = ((chunk % config_mrsg.mrsg_number_of_workers)
69  + ((config_mrsg.mrsg_number_of_workers / config_mrsg.mrsg_chunk_replicas) * r)
70  ) % config_mrsg.mrsg_number_of_workers;
71 
72  chunk_owner_mrsg[chunk][owner] = 1;
73  }
74  }
75  }
76 }

78 size_t find_random_chunk_owner_mrsg (int cid)
79 {
80  int replica;
81  size_t owner = NONE;
82  size_t mrsg_wid;
83 
84  replica = rand () % config_mrsg.mrsg_chunk_replicas;
85 
86  for (mrsg_wid = 0; mrsg_wid < config_mrsg.mrsg_number_of_workers; mrsg_wid++)
87  {
88  if (chunk_owner_mrsg[cid][mrsg_wid])
89  {
90  owner = mrsg_wid;
91 
92  if (replica == 0)
93  break;
94  else
95  replica--;
96  }
97  }
98 
99  xbt_assert (owner != NONE, "Aborted: chunk %d is missing.", cid);
100 
101  return owner;
102 }

154  MSG_task_dsend (MSG_task_create ("DATA-IP", 0.0, data_size, NULL), mailbox, NULL);
