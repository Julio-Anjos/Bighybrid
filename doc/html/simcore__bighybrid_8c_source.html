00329 static void init_mra_config (void)
00330 {
00331     const char*    process_name = NULL;
00332     msg_host_t     host;
00333     msg_process_t  process;
00334     size_t         mra_wid;
00335     unsigned int   cursor;
00336     w_mra_info_t       wi;
00337     xbt_dynar_t    process_list;
00338 
00339     /* Initialize MRA hosts information. */
00340 
00341     config_mra.mra_number_of_workers = 0;
00342 
00343     process_list = MSG_processes_as_dynar ();
00344     xbt_dynar_foreach (process_list, cursor, process)
00345     {
00346         process_name = MSG_process_get_name (process);
00347         if ( strcmp (process_name, "worker_mra") == 0 )
00348             config_mra.mra_number_of_workers++;
00349     }
00350 
00351     config_mra.workers_mra = xbt_new (msg_host_t, config_mra.mra_number_of_workers);
00352 
00353     mra_wid = 0;
00354     config_mra.grid_cpu_power = 0.0;
00355     xbt_dynar_foreach (process_list, cursor, process)
00356     {
00357         process_name = MSG_process_get_name (process);
00358         host = MSG_process_get_host (process);
00359         if ( strcmp (process_name, "worker_mra") == 0 )
00360         {
00361             config_mra.workers_mra[mra_wid] = host;
00362             /* Set the worker ID as its data. */
00363             wi = xbt_new (struct mra_w_info_s, 1);
00364             wi->mra_wid = mra_wid;
00365             MSG_host_set_data (host, (void*)wi);
00366             /* Add the worker's cpu power to the grid total. */
00367             config_mra.grid_cpu_power += MSG_get_host_speed (host);
00368             mra_wid++;
00369         }
00370     }
00371     config_mra.grid_average_speed = config_mra.grid_cpu_power / config_mra.mra_number_of_workers;
00372     config_mra.mra_heartbeat_interval = mra_maxval (MRA_HEARTBEAT_MIN_INTERVAL, config_mra.mra_number_of_workers / 100);
00373     config_mra.amount_of_tasks_mra[MRA_MAP] = config_mra.mra_chunk_count;
00374     config_mra.initialized = 1;
00375 }
