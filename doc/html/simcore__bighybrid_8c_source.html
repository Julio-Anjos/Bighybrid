224 * @brief Reads config file and creates a vector to each node with the availability type, and start and end times.
225 */
226 static int read_mra_vc_config_file (const char* vc_file_name, int n_line)
227 {
228  int i;
229  /* Allocate memory for the volunteer computing matrix. */
230  vc_node = xbt_new(int*, (n_line * sizeof (int)));
231  vc_type = xbt_new(int*, (n_line * sizeof (int)));
232  vc_start = xbt_new(long double*, (n_line * sizeof (long double)));
233  vc_end = xbt_new(long double*, (n_line * sizeof (long double)));
234  /* Allocate memory for the elements of volunteer computing matrix. */
235  for (i = 0; i < (n_line + 1); i++)
236  {
237  vc_node[i] = xbt_new(int, (sizeof (int)));
238  vc_type[i] = xbt_new(int, (sizeof (int)));
239  vc_start[i] = xbt_new(long double, (sizeof (long double)));
240  vc_end[i] = xbt_new(long double, (sizeof (long double)));
241  }
242 
243  VC_TRACE *ptr_one;
244  FILE* vc_file;
245 
246  ptr_one = (VC_TRACE *) malloc(n_line * sizeof(VC_TRACE));
247  vc_file = fopen (vc_file_name, "r");
248 
249  i=0;
250  while(!feof(vc_file))
251  {
252  fscanf(vc_file, "%d,%d,%Lf,%Lf", &ptr_one->mra_vc_node_id, &ptr_one->mra_vc_status_type,
253  &ptr_one->mra_vc_start, &ptr_one->mra_vc_end);
254 
255  vc_node[i][0] = ptr_one->mra_vc_node_id;
256  vc_type[i][0] = ptr_one->mra_vc_status_type;
257  vc_start[i][0] = ptr_one->mra_vc_start;
258  vc_end[i][0]= ptr_one->mra_vc_end;
259  i++;
260  }
261 
262  return 0;
263 
264  fclose(vc_file);
265 }

295 static void read_bighybrid_config_file (const char* file_name)
296 {
297  char property[256];
298  FILE* file;
299 
300  /* Set the default MRA configuration. */
301  config_mra.mra_chunk_size = 67108864;
302  config_mra.mra_chunk_count = 0;
303  config_mra.mra_chunk_replicas = 3;
304  config_mra.mra_slots[MRA_MAP] = 2;
305  config_mra.amount_of_tasks_mra[MRA_REDUCE] = 1;
306  config_mra.mra_slots[MRA_REDUCE] = 2;
307  config_mra.Fg=1;
308  config_mra.mra_perc=100;
309 
310  /* Set the default MRSG configuration. */
311  config_mrsg.mrsg_chunk_size = 67108864;
312  config_mrsg.mrsg_chunk_count = 0;
313  config_mrsg.mrsg_chunk_replicas = 3;
314  config_mrsg.mrsg_slots[MRSG_MAP] = 2;
315  config_mrsg.amount_of_tasks_mrsg[MRSG_REDUCE] = 1;
316  config_mrsg.mrsg_slots[MRSG_REDUCE] = 2;
317  config_mrsg.mrsg_perc = 100;
318 
319  /* Read the user configuration file. */
320 
321  file = fopen (file_name, "r");
322 
323  xbt_assert (file != NULL, "Error reading cofiguration file: %s", file_name);
324 
325  while ( fscanf (file, "%256s", property) != EOF )
326  {
327  if ( strcmp (property, "mra_chunk_size") == 0 )
328  {
329  fscanf (file, "%lg", &config_mra.mra_chunk_size);
330  config_mra.mra_chunk_size *= 1024 * 1024; /* MB -> bytes */
331  }
332  else if ( strcmp (property, "mra_input_chunks") == 0 )

344  else if ( strcmp (property, "grain_factor") == 0 )
345  {
346  fscanf (file, "%d", &config_mra.Fg);
347  }
348  else if ( strcmp (property, "mra_intermed_perc") == 0 )
349  {
350  fscanf (file, "%lg", &config_mra.mra_perc);

352  else if ( strcmp (property, "mra_reduces") == 0 )

355  }

357  {
358  fscanf (file, "%d", &config_mra.mra_slots[MRA_REDUCE]);
359  }
360  else if ( strcmp (property, "perc_num_volatile_node") == 0 )

362  fscanf (file, "%lg", &config_mra.perc_vc_node);
363  }
364  else if ( strcmp (property, "failure_timeout") == 0 )
365  {
366  fscanf (file, "%lg", &config_mra.failure_timeout_conf);

368  else if ( strcmp (property, "mrsg_chunk_size") == 0 )

370  fscanf (file, "%lg", &config_mrsg.mrsg_chunk_size);
371  config_mrsg.mrsg_chunk_size *= 1024 * 1024; /* MB -> bytes */
372  }
373  else if ( strcmp (property, "mrsg_input_chunks") == 0 )

375  fscanf (file, "%d", &config_mrsg.mrsg_chunk_count);

382  {
383  fscanf (file, "%d", &config_mrsg.mrsg_slots[MRSG_MAP]);
384  }
385  else if ( strcmp (property, "mrsg_reduces") == 0 )
386  {
387  fscanf (file, "%d", &config_mrsg.amount_of_tasks_mrsg[MRSG_REDUCE]);
388  }
389  else if ( strcmp (property, "mrsg_reduce_slots") == 0 )
390  {
391  fscanf (file, "%d", &config_mrsg.mrsg_slots[MRSG_REDUCE]);
392  }
393  else if ( strcmp (property, "mrsg_intermed_perc") == 0 )
394  {
395  fscanf (file, "%lg", &config_mrsg.mrsg_perc);

397  else

405

407

411  xbt_assert (config_mra.mra_slots[MRA_MAP] > 0, "MRA_Map slots must be greater than zero");
412  xbt_assert (config_mra.amount_of_tasks_mra[MRA_REDUCE] >= 0, "The number of MRA_reduce tasks can't be negative");

431 static void init_mra_config (void)
432 {
433  const char* process_name = NULL;
434  msg_host_t host;
435  msg_process_t process;
436  size_t mra_wid;
437  unsigned int cursor;
438  w_mra_info_t wi;
439  xbt_dynar_t process_list;
440 
441  /* Initialize MRA hosts information. */
442 
443  config_mra.mra_number_of_workers = 0;
444 
445  process_list = MSG_processes_as_dynar ();
446  xbt_dynar_foreach (process_list, cursor, process)
447  {
448  process_name = MSG_process_get_name (process);
449  if ( strcmp (process_name, "worker_mra") == 0 )
450  config_mra.mra_number_of_workers++;
451  }
452 
453  config_mra.workers_mra = xbt_new (msg_host_t, config_mra.mra_number_of_workers);
454 
455  mra_wid = 0;
456  config_mra.grid_cpu_power = 0.0;
457  xbt_dynar_foreach (process_list, cursor, process)
458  {
459  process_name = MSG_process_get_name (process);
460  host = MSG_process_get_host (process);
461  if ( strcmp (process_name, "worker_mra") == 0 )
462  {
463  config_mra.workers_mra[mra_wid] = host;
464  /* Set the worker ID as its data. */
465  wi = xbt_new (struct mra_w_info_s, 1);
466  wi->mra_wid = mra_wid;
467  MSG_host_set_data (host, (void*)wi);
468  /* Add the worker's cpu power to the grid total. */
469  config_mra.grid_cpu_power += MSG_get_host_speed (host);
470  mra_wid++;
471  }
472  }
473  config_mra.grid_average_speed = config_mra.grid_cpu_power / config_mra.mra_number_of_workers;
474  config_mra.mra_heartbeat_interval = mra_maxval (MRA_HEARTBEAT_MIN_INTERVAL, config_mra.mra_number_of_workers / 100);
475  config_mra.amount_of_tasks_mra[MRA_MAP] = config_mra.mra_chunk_count;
476  config_mra.initialized = 1;
477 }

534 static void init_job_mra (void)
535 {
536  int i;
537  size_t mra_wid;
538 
539  xbt_assert (config_mra.initialized, "init_mra_config has to be called before init_job_mra");
540 
541  job_mra.finished = 0;
542  job_mra.mra_heartbeats = xbt_new (struct mra_heartbeat_s, config_mra.mra_number_of_workers);
543  for (mra_wid = 0; mra_wid < config_mra.mra_number_of_workers; mra_wid++)
544  {
545  job_mra.mra_heartbeats[mra_wid].slots_av[MRA_MAP] = config_mra.mra_slots[MRA_MAP];
546  job_mra.mra_heartbeats[mra_wid].slots_av[MRA_REDUCE] = config_mra.mra_slots[MRA_REDUCE];
547  job_mra.mra_heartbeats[mra_wid].wid_timestamp = sizeof(long double);
548  }
549 
550  /* Initialize map information. */
551  job_mra.tasks_pending[MRA_MAP] = config_mra.amount_of_tasks_mra[MRA_MAP];
552  job_mra.task_status[MRA_MAP] = xbt_new0 (int, config_mra.amount_of_tasks_mra[MRA_MAP]);
553  job_mra.task_instances[MRA_MAP] = xbt_new0 (int, config_mra.amount_of_tasks_mra[MRA_MAP]);
554  job_mra.task_list[MRA_MAP] = xbt_new0 (msg_task_t*, config_mra.amount_of_tasks_mra[MRA_MAP]);
555  for (i = 0; i < config_mra.amount_of_tasks_mra[MRA_MAP]; i++)
556  job_mra.task_list[MRA_MAP][i] = xbt_new0 (msg_task_t, MAX_SPECULATIVE_COPIES);
557 
558  /* Initialize Reduce tasks number. */
559  if (config_mra.Fg != 1) {
560  config_mra.amount_of_tasks_mra[MRA_REDUCE] = config_mra.Fg * config_mra.mra_number_of_workers;
561  }
562
