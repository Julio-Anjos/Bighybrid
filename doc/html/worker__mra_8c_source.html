107 static int mra_vc_sleep_f (size_t my_id, double vc_time_stamp)
108 {
109  int j=0;
110  int i=0;
111  int hosts_vc_traces;
112  int mra_vc_job_hosts;
113 
114  /* Number of hosts from trace archive*/
115  hosts_vc_traces = vc_node[config_mra_vc_file_line[0]][0];
116  /*Number of volatile hosts defined by user. The result is saved on element array config_mra_vc_file_line[1]*/
117  mra_vc_job_hosts = (int)(ceil(config_mra.mra_number_of_workers * (double)config_mra.perc_vc_node/100)) ;
118  config_mra_vc_file_line[1] = mra_vc_job_hosts;
119  if ( my_id > hosts_vc_traces && mra_vc_job_hosts > hosts_vc_traces ){
120  XBT_INFO ("Atention - Number of volatile host %d, in log file is insufficient to simulation \n", hosts_vc_traces);
121  return 0;
122 
123  }
124 
125  if ( my_id < config_mra_vc_file_line[1])
126  {
127  for (i=0; i < config_mra_vc_file_line[0] ; i++)
128  {
129  if ((my_id + 1 == vc_node[i][0]) && my_id < mra_vc_job_hosts )
130  {
131  j=i;
132  while ((j < config_mra_vc_file_line[0] ))
133  {
134  if (vc_time_stamp >= vc_start[j][0] && vc_time_stamp <= vc_end[j][0])
135  {
136  if (vc_type[j][0] == 1)
137  {
138  vc_traces_time = config_mra.mra_heartbeat_interval;
139 
140  job_mra.mra_heartbeats[my_id].wid_timestamp = MSG_get_clock ();
141  //XBT_INFO (" Volat_node %zd ON - Traces_time %g, Hearbeat %Lg \n",my_id,vc_traces_time, job_mra.mra_heartbeats[my_id].wid_timestamp);
142  return vc_traces_time;
143  break;
144  }
145  else
146  {
147  vc_traces_time = (vc_end[j][0] - vc_time_stamp);
148  if (config_mra.mra_heartbeat_interval < vc_traces_time)
149  {
150  job_mra.mra_heartbeats[my_id].wid_timestamp = MSG_get_clock ();
151  }
152  //XBT_INFO (" Volat_node %zd OFF - Traces_time %g, Hearbeat %Lg \n", my_id,vc_traces_time, job_mra.mra_heartbeats[my_id].wid_timestamp);
153  return vc_traces_time;
154  break;
155  }
156  }
157  j++;
158  }
159  }
160  }
161  }
162  else
163  {
164  vc_traces_time = config_mra.mra_heartbeat_interval;
165  job_mra.mra_heartbeats[my_id].wid_timestamp = MSG_get_clock ();
166  //XBT_INFO (" Host %zd ON - Traces_time %g, Hearbeat %Lg \n", my_id, vc_traces_time, job_mra.mra_heartbeats[my_id].wid_timestamp);
167  }
168  return vc_traces_time;

179 static int listen_mra (int argc, char* argv[])
180 {
181  char mailbox[MAILBOX_ALIAS_SIZE];
182  msg_error_t status;
183  msg_host_t me;
184  msg_task_t msg = NULL;
185 
186  me = MSG_host_self ();
187  sprintf (mailbox, TASKTRACKER_MRA_MAILBOX, get_mra_worker_id (me));
188 
189  while (!job_mra.finished)
190  {
191  msg = NULL;
192  status = receive (&msg, mailbox);
193 
194  if (status == MSG_OK && mra_message_is (msg, SMS_TASK_MRA))
195  {
196  MSG_process_create ("compute_mra", compute_mra, msg, me);
197  }
198  else if (mra_message_is (msg, SMS_FINISH_MRA))
199  {
200  MSG_task_destroy (msg);
201  break;
202  }
203  }
204 
205  return 0;
206 }

211 static int compute_mra (int argc, char* argv[])
212 {
213  msg_error_t status;
214  msg_task_t mra_task;
215  mra_task_info_t ti;
216  xbt_ex_t e;
217 
218  mra_task = (msg_task_t) MSG_process_get_data (MSG_process_self ());
219  ti = (mra_task_info_t) MSG_task_get_data (mra_task);
220  ti->mra_pid = MSG_process_self_PID ();
221 
222 
223  switch (ti->mra_phase)
224  {
225  case MRA_MAP:
226  get_mra_chunk (ti);
227  mra_task_ftm[ti->mra_tid].mra_ft_pid[MRA_MAP] = ti->mra_pid;
228  //XBT_INFO ("INFO:Processo Map %d - Tarefa %zd", mra_task_ftm[ti->mra_tid].mra_ft_pid[MRA_MAP], ti->mra_tid );
229  break;
230 
231  case MRA_REDUCE:
232  get_mra_map_output (ti);
233  mra_task_ftm[ti->mra_tid].mra_ft_pid[MRA_REDUCE] = ti->mra_pid;
234  //XBT_INFO ("INFO:Processo Reduce %d - Tarefa %zd", mra_task_ftm[ti->mra_tid].mra_ft_pid[MRA_REDUCE],ti->mra_tid );
235  break;
236  }
237 
238  if (job_mra.task_status[ti->mra_phase][ti->mra_tid] != T_STATUS_MRA_DONE)
239  {
240  TRY
241  {
242  status = MSG_task_execute (mra_task);
243 
244  if (ti->mra_phase == MRA_MAP && status == MSG_OK)
245  update_mra_map_output (MSG_host_self (), ti->mra_tid);
246  }
247  CATCH (e)
248  {
249  xbt_assert (e.category == cancel_error, "%s", e.msg);
250  xbt_ex_free (e);
251  }
252  }
253 
254  job_mra.mra_heartbeats[ti->mra_wid].slots_av[ti->mra_phase]++;
255 
256  if (!job_mra.finished)
257  send (SMS_TASK_MRA_DONE, 0.0, 0.0, ti, MASTER_MRA_MAILBOX);
258 
259  return 0;
260 }

310 static void get_mra_map_output (mra_task_info_t ti)
311 {
312  char mailbox[MAILBOX_ALIAS_SIZE];
313  msg_error_t status;
314  msg_task_t data = NULL;
315  size_t total_copied, must_copy;
316  size_t my_id;
317  size_t mra_wid;
318  size_t* data_copied;
319 
320  my_id = get_mra_worker_id (MSG_host_self ());
321  data_copied = xbt_new0 (size_t, config_mra.mra_number_of_workers);
322  ti->map_output_copied = data_copied;
323  total_copied = 0;
324  must_copy = reduce_mra_input_size (ti->mra_tid);
325 
326 
327 #ifdef VERBOSE
328  XBT_INFO ("INFO: start copy");
329 #endif
330  while (total_copied < must_copy)
331  {
332  for (mra_wid = 0; mra_wid < config_mra.mra_number_of_workers; mra_wid++)
333  {
334  if (job_mra.task_status[MRA_REDUCE][ti->mra_tid] == T_STATUS_MRA_DONE)
335  {
336  xbt_free_ref (&data_copied);
337  return;
338  }
339 
340  if (job_mra.map_output[mra_wid][ti->mra_tid] > data_copied[mra_wid])
341  {
342  sprintf (mailbox, DATANODE_MRA_MAILBOX, mra_wid);
343  status = send (SMS_GET_INTER_MRA_PAIRS, 0.0, 0.0, ti, mailbox);
344  if (status == MSG_OK)
345  {
346  sprintf (mailbox, TASK_MRA_MAILBOX, my_id, MSG_process_self_PID ());
347  data = NULL;
348  //TODO Set a timeout: reduce.copy.backoff
349  status = receive (&data, mailbox);
350  if (status == MSG_OK)
351  {
352  data_copied[mra_wid] += MSG_task_get_data_size (data);
353  total_copied += MSG_task_get_data_size (data);
354  MSG_task_destroy (data);
355  }
356  }
357  }
358  }
359  /* (Hadoop 0.20.2) mapred/ReduceTask.java:1979 */
360  MSG_process_sleep (5);
361  }
362 
363 
364 #ifdef VERBOSE
365  XBT_INFO ("INFO: copy finished");
366 #endif
367  ti->shuffle_mra_end = MSG_get_clock ();
368 
369  xbt_free_ref (&data_copied);
370 }
