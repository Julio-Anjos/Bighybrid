42 int master_mrsg (int argc, char* argv[])
43 {
44  mrsg_heartbeat_t mrsg_heartbeat;
45  msg_error_t status;
46  msg_host_t worker_mrsg;
47  msg_task_t msg = NULL;
48  size_t mrsg_wid;
49  mrsg_task_info_t ti;
50 
51  print_mrsg_config ();
52  XBT_INFO ("JOB BEGIN"); XBT_INFO (" ");
53 
54 
55  tasks_log = fopen ("tasks-mrsg.csv", "w");
56  fprintf (tasks_log, "task_id,mrsg_phase,worker_id,time,action,shuffle_end\n");
57 
58  while (job_mrsg.tasks_pending[MRSG_MAP] + job_mrsg.tasks_pending[MRSG_REDUCE] > 0)
59  {
60  msg = NULL;
61  status = receive (&msg, MASTER_MRSG_MAILBOX);
62  if (status == MSG_OK)
63  {
64  worker_mrsg = MSG_task_get_source (msg);
65  mrsg_wid = get_mrsg_worker_id (worker_mrsg);
66 
67  if (mrsg_message_is (msg, SMS_HEARTBEAT_MRSG))
68  {
69  mrsg_heartbeat = &job_mrsg.mrsg_heartbeats[mrsg_wid];
70  if (is_straggler_mrsg (worker_mrsg))
71  {
72  set_mrsg_speculative_tasks (worker_mrsg);
73  }
74  else
75  {
76  if (mrsg_heartbeat->slots_av[MRSG_MAP] > 0)
77  send_map_to_mrsg_worker (worker_mrsg);
78 
79  if (mrsg_heartbeat->slots_av[MRSG_REDUCE] > 0)
80  send_reduce_to_mrsg_worker (worker_mrsg);
81  }
82  }
83  else if (mrsg_message_is (msg, SMS_TASK_MRSG_DONE))
84  {
85  ti = (mrsg_task_info_t) MSG_task_get_data (msg);
86 
87  if (job_mrsg.task_status[ti->mrsg_phase][ti->mrsg_tid] != T_STATUS_MRSG_DONE)
88  {
89  job_mrsg.task_status[ti->mrsg_phase][ti->mrsg_tid] = T_STATUS_MRSG_DONE;
90  finish_all_mrsg_task_copies (ti);
91  job_mrsg.tasks_pending[ti->mrsg_phase]--;
92  if (job_mrsg.tasks_pending[ti->mrsg_phase] <= 0)
93  {
94  XBT_INFO (" ");
95  XBT_INFO ("%s PHASE DONE", (ti->mrsg_phase==MRSG_MAP?"MRSG_MAP":"MRSG_REDUCE"));
96  XBT_INFO (" ");
97  }
98  }
99  xbt_free_ref (&ti);
100  }
101  MSG_task_destroy (msg);
102  }
103  }
104 
105  fclose (tasks_log);
106 
107  job_mrsg.finished = 1;
108 
109  print_mrsg_config ();
110  print_mrsg_stats ();
111  XBT_INFO ("JOB END");
112 
113  return 0;
114 }

155 static int is_straggler_mrsg (msg_host_t worker_mrsg)

159 
160  mrsg_wid = get_mrsg_worker_id (worker_mrsg);
161

163

232 static void send_map_to_mrsg_worker (msg_host_t dest)
233 {
234  char* flags;
235  int task_type;
236  size_t chunk;
237  size_t sid = NONE;
238  size_t tid = NONE;
239  size_t mrsg_wid;
240 
241  if (job_mrsg.tasks_pending[MRSG_MAP] <= 0)
242  return;
243 
244  enum { LOCAL, REMOTE, LOCAL_SPEC, REMOTE_SPEC, NO_TASK };
245  task_type = NO_TASK;
246 
247  mrsg_wid = get_mrsg_worker_id (dest);
248 
249  /* Look for a task for the worker. */
250  for (chunk = 0; chunk < config_mrsg.mrsg_chunk_count; chunk++)
251  {
252  if (job_mrsg.task_status[MRSG_MAP][chunk] == T_STATUS_MRSG_PENDING)
253  {
254  if (chunk_owner_mrsg[chunk][mrsg_wid])
255  {
256  task_type = LOCAL;
257  tid = chunk;
258  break;
259  }
260  else
261  {
262  task_type = REMOTE;
263  tid = chunk;
264  }
265  }
266  else if (job_mrsg.task_status[MRSG_MAP][chunk] == T_STATUS_MRSG_TIP_SLOW
267  && task_type > REMOTE
268  && job_mrsg.task_instances[MRSG_MAP][chunk] < 2)
269  {
270  if (chunk_owner_mrsg[chunk][mrsg_wid])
271  {
272  task_type = LOCAL_SPEC;
273  tid = chunk;
274  }
275  else if (task_type > LOCAL_SPEC)
276  {
277  task_type = REMOTE_SPEC;
278  tid = chunk;
279  }
280  }
281  }
282 
283  switch (task_type)
284  {
285  case LOCAL:
286  flags = "";
287  sid = mrsg_wid;
288  stats_mrsg.map_local_mrsg++;
289  break;
290 
291  case REMOTE:
292  flags = "(non-local)";
293  sid = find_random_chunk_owner_mrsg (tid);
294  stats_mrsg.map_remote_mrsg++;
295  break;
296 
297  case LOCAL_SPEC:
298  flags = "(speculative)";
299  sid = mrsg_wid;
300  stats_mrsg.map_spec_mrsg_l++;
301  break;
302 
303  case REMOTE_SPEC:
304  flags = "(non-local, speculative)";
305  sid = find_random_chunk_owner_mrsg (tid);
306  stats_mrsg.map_spec_mrsg_r++;
307  break;
308 
309  default: return;
310  }
311 
312  XBT_INFO ("map %zu assigned to %s %s", tid, MSG_host_get_name (dest), flags);
313 
314  send_mrsg_task (MRSG_MAP, tid, sid, dest);
315 }

321 static void send_reduce_to_mrsg_worker (msg_host_t dest)
322 {
323  char* flags;
324  int task_type;
325  size_t t;
326  size_t tid = NONE;
327

338  task_type = NORMAL;
339  tid = t;

343  && job_mrsg.task_instances[MRSG_REDUCE][t] < 2)
344  {
345  task_type = SPECULATIVE;
346  tid = t;
347  }
348  }
349

351  {
352  case NORMAL:
353  flags = "";
354  stats_mrsg.reduce_mrsg_normal++;
355  break;
356 
357  case SPECULATIVE:
358  flags = "(speculative)";
359  stats_mrsg.reduce_mrsg_spec++;

365  XBT_INFO ("reduce %zu assigned to %s %s", tid, MSG_host_get_name (dest), flags);
366 
367  send_mrsg_task (MRSG_REDUCE, tid, NONE, dest);

377 static void send_mrsg_task (enum mrsg_phase_e mrsg_phase, size_t tid, size_t data_src, msg_host_t dest)
378 {
379  char mailbox[MAILBOX_ALIAS_SIZE];
380  int i;
381  double cpu_required = 0.0;
382  msg_task_t mrsg_task = NULL;
383  mrsg_task_info_t task_info;
384  size_t mrsg_wid;
385 
386  mrsg_wid = get_mrsg_worker_id (dest);
387 
388  cpu_required = user_mrsg.task_cost_f (mrsg_phase, tid, mrsg_wid);
389 
390  task_info = xbt_new (struct mrsg_task_info_s, 1);
391  mrsg_task = MSG_task_create (SMS_TASK_MRSG, cpu_required, 0.0, (void*) task_info);
392 
393  task_info->mrsg_phase = mrsg_phase;
394  task_info->mrsg_tid = tid;
395  task_info->mrsg_src = data_src;
396  task_info->mrsg_wid = mrsg_wid;
397  task_info->mrsg_task = mrsg_task;
398  task_info->shuffle_mrsg_end = 0.0;
399 
400  // for tracing purposes...
401  MSG_task_set_category (mrsg_task, (mrsg_phase==MRSG_MAP?"MRSG_MAP":"MRSG_REDUCE"));
402 
403  if (job_mrsg.task_status[mrsg_phase][tid] != T_STATUS_MRSG_TIP_SLOW)
404  job_mrsg.task_status[mrsg_phase][tid] = T_STATUS_MRSG_TIP;
405 
406  job_mrsg.mrsg_heartbeats[mrsg_wid].slots_av[mrsg_phase]--;
407 
408  for (i = 0; i < MAX_SPECULATIVE_COPIES; i++)
409  {
410  if (job_mrsg.task_list[mrsg_phase][tid][i] == NULL)
411  {
412  job_mrsg.task_list[mrsg_phase][tid][i] = mrsg_task;
413  break;
414  }
415  }
416 
417  fprintf (tasks_log, "%d_%zu_%d,%s,%zu,%.3f,START,\n", mrsg_phase, tid, i, (mrsg_phase==MRSG_MAP?"MRSG_MAP":"MRSG_REDUCE"), mrsg_wid, MSG_get_clock ());
418 
419 #ifdef VERBOSE
420  XBT_INFO ("TX: %s > %s", SMS_TASK_MRSG, MSG_host_get_name (dest));
421 #endif
422 
423  sprintf (mailbox, TASKTRACKER_MRSG_MAILBOX, mrsg_wid);
424  xbt_assert (MSG_task_send (mrsg_task, mailbox) == MSG_OK, "ERROR SENDING MESSAGE");
425 
426  job_mrsg.task_instances[mrsg_phase][tid]++;
427 }
