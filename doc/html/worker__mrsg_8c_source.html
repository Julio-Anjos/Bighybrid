78 static int listen_mrsg (int argc, char* argv[])
79 {
80  char mailbox[MAILBOX_ALIAS_SIZE];
81  msg_error_t status;
82  msg_host_t me;
83  msg_task_t msg = NULL;
84 
85  me = MSG_host_self ();
86  sprintf (mailbox, TASKTRACKER_MRSG_MAILBOX, get_mrsg_worker_id (me));
87 
88  while (!job_mrsg.finished)
89  {
90  msg = NULL;
91  status = receive (&msg, mailbox);
92 
93  if (status == MSG_OK && mrsg_message_is (msg, SMS_TASK_MRSG))
94  {
95  MSG_process_create ("compute_mrsg", compute_mrsg, msg, me);
96  }
97  else if (mrsg_message_is (msg, SMS_FINISH_MRSG))
98  {
99  MSG_task_destroy (msg);
100  break;
101  }
102  }
103 
104  return 0;
105 }
109 static int compute_mrsg (int argc, char* argv[])
110 {
111  msg_error_t status;
112  msg_task_t mrsg_task;
113  mrsg_task_info_t ti;
114  xbt_ex_t e;
115 
116  mrsg_task = (msg_task_t) MSG_process_get_data (MSG_process_self ());
117  ti = (mrsg_task_info_t) MSG_task_get_data (mrsg_task);
118  ti->mrsg_pid = MSG_process_self_PID ();
119 
120  switch (ti->mrsg_phase)
121  {
122  case MRSG_MAP:
123  get_mrsg_chunk (ti);
124  break;
125 
126  case MRSG_REDUCE:
127  get_mrsg_map_output (ti);
128  break;
129  }
130 
131  if (job_mrsg.task_status[ti->mrsg_phase][ti->mrsg_tid] != T_STATUS_MRSG_DONE)
132  {
133  TRY
134  {
135  status = MSG_task_execute (mrsg_task);
136 
137  if (ti->mrsg_phase == MRSG_MAP && status == MSG_OK)
138  update_mrsg_map_output (MSG_host_self (), ti->mrsg_tid);
139  }
140  CATCH (e)
141  {
142  xbt_assert (e.category == cancel_error, "%s", e.msg);
143  xbt_ex_free (e);
144  }
145  }
146 
147  job_mrsg.mrsg_heartbeats[ti->mrsg_wid].slots_av[ti->mrsg_phase]++;
148 
149  if (!job_mrsg.finished)
150  send (SMS_TASK_MRSG_DONE, 0.0, 0.0, ti, MASTER_MRSG_MAILBOX);
151 
152  return 0;
153 }

174 {

182  /* Request the chunk to the source node. */
183  if (ti->mrsg_src != my_id)
184  {
185  sprintf (mailbox, DATANODE_MRSG_MAILBOX, ti->mrsg_src);

187  if (status == MSG_OK)

190  status = receive (&data, mailbox);

193  }
194  }

200 static void get_mrsg_map_output (mrsg_task_info_t ti)
201 {
202  char mailbox[MAILBOX_ALIAS_SIZE];
203  msg_error_t status;
204  msg_task_t data = NULL;
205  size_t total_copied, must_copy;
206  size_t my_id;
207  size_t mrsg_wid;
208  size_t* data_copied;
209 
210  my_id = get_mrsg_worker_id (MSG_host_self ());
211  data_copied = xbt_new0 (size_t, config_mrsg.mrsg_number_of_workers);
212  ti->map_output_copied = data_copied;
213  total_copied = 0;
214  must_copy = reduce_mrsg_input_size (ti->mrsg_tid);
215 
216 #ifdef VERBOSE
217  XBT_INFO ("INFO: start copy");
218 #endif
219 
220  while (total_copied < must_copy)
221  {
222  for (mrsg_wid = 0; mrsg_wid < config_mrsg.mrsg_number_of_workers; mrsg_wid++)
223  {
224  if (job_mrsg.task_status[MRSG_REDUCE][ti->mrsg_tid] == T_STATUS_MRSG_DONE)
225  {
226  xbt_free_ref (&data_copied);
227  return;
228  }
229 
230  if (job_mrsg.map_output[mrsg_wid][ti->mrsg_tid] > data_copied[mrsg_wid])
231  {
232  sprintf (mailbox, DATANODE_MRSG_MAILBOX, mrsg_wid);
233  status = send (SMS_GET_INTER_MRSG_PAIRS, 0.0, 0.0, ti, mailbox);
234  if (status == MSG_OK)
235  {
236  sprintf (mailbox, TASK_MRSG_MAILBOX, my_id, MSG_process_self_PID ());
237  data = NULL;
238  //TODO Set a timeout: reduce.copy.backoff
239  status = receive (&data, mailbox);
240  if (status == MSG_OK)
241  {
242  data_copied[mrsg_wid] += MSG_task_get_data_size (data);
243  total_copied += MSG_task_get_data_size (data);
244  MSG_task_destroy (data);
245  }
246  }
247  }
248  }
249  /* (Hadoop 0.20.2) mapred/ReduceTask.java:1979 */
250  MSG_process_sleep (5);
251  }
252 #ifdef VERBOSE
253  XBT_INFO ("INFO: copy finished");
254 #endif
255  ti->shuffle_mrsg_end = MSG_get_clock ();
256 
257  xbt_free_ref (&data_copied);
258 }
