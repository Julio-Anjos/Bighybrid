26 static void read_mra_config_file (const char* file_name)
27 {
28  char property[256];
29  FILE* file;
30 
31  /* Set the default configuration. */
32  config_mra.mra_chunk_size = 67108864;
33  config_mra.amount_of_tasks_mra[MRA_REDUCE] = 1;
34  config_mra.Fg=1;
35  config_mra.mra_perc=100;
36 
37  /* Read the user configuration file. */
38 
39  file = fopen (file_name, "r");
40  /* Read the user configuration file. */
41 
42  xbt_assert (file != NULL, "Error reading cofiguration file: %s", file_name);
43 
44  while ( fscanf (file, "%256s", property) != EOF )
45  {
46  if ( strcmp (property, "mra_chunk_size") == 0 )
47  {
48  fscanf (file, "%lg", &config_mra.mra_chunk_size);
49  config_mra.mra_chunk_size *= 1024 * 1024; /* MB -> bytes */
50  }
51  else if ( strcmp (property, "grain_factor") == 0 )
52  {
53  fscanf (file, "%d", &config_mra.Fg);
54  }
55  else if ( strcmp (property, "mra_intermed_perc") == 0 )
56  {
57  fscanf (file, "%g", &config_mra.mra_perc);
58  }
59  else if ( strcmp (property, "mra_reduces") == 0 )
60  {
61  fscanf (file, "%d", &config_mra.amount_of_tasks_mra[MRA_REDUCE]);
62  }
63  else
64  {
65  printf ("Error: Property %s is not valid. (in %s)", property, file_name);
66  exit (1);
67  }
68 
69  }
70 
71  fclose (file);
72 
73 }
