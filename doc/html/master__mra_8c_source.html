279 static void mra_status_tfm (enum mra_phase_e mra_phase, size_t ft_task_id, size_t ft_mra_wid, enum mra_task_status_e mra_ft_task_status )
280 {
281  size_t k = 0 ;
282 
283  switch (mra_phase)
284  {
285  case MRA_MAP:
286  k = ft_task_id;
287  break;
288  case MRA_REDUCE:
289  k = ft_task_id + config_mra.amount_of_tasks_mra[MRA_MAP];
290  break;
291  }
292  if (mra_ft_task_status == T_STATUS_MRA_DONE && mra_phase == mra_task_ftm[k].mra_ft_phase)
293  {
294  mra_ftsys_f.mra_ft_task_status = mra_ft_task_status;
295  mra_task_ftm[k].mra_ft_task_status = mra_ftsys_f.mra_ft_task_status;
296  }
297 
298  else
299  {
300  /*Task Phase*/
301  mra_ftsys_f.mra_ft_phase = mra_phase;
302  mra_task_ftm[k].mra_ft_phase = mra_ftsys_f.mra_ft_phase;
303  /*Task Status*/
304  mra_ftsys_f.mra_ft_task_status = mra_ft_task_status;
305  mra_task_ftm[k].mra_ft_task_status = mra_ftsys_f.mra_ft_task_status;
306  /*Task Work_id*/
307  mra_ftsys_f.mra_ft_wid = ft_mra_wid;
308  mra_task_ftm[k].mra_ft_wid = mra_ftsys_f.mra_ft_wid;
309  /*Task Number*/
310  if (mra_phase == MRA_MAP)
311  {
312  mra_ftsys_f.mra_ft_task_id = ft_task_id;
313  mra_task_ftm[k].mra_ft_task_id = mra_ftsys_f.mra_ft_task_id;
314  }
315  else
316  {
317  mra_ftsys_f.mra_ft_task_id = ft_task_id;
318  mra_task_ftm[k].mra_ft_task_id = mra_ftsys_f.mra_ft_task_id;
319  }
320  /*Dispatch Task*/
321  mra_ftm_done_s[k].mra_ft_task_status = T_STATUS_MRA_DISP;
322 
323  }

394 static void get_vc_behavior (double control_timestamp, size_t mra_wid, double last_hb_wid_timestamp )
395 {
396  int i=0;
397 
398 
399  while (i < config_mra.mra_number_of_workers )
400  {
401  if(mra_wid == i)
402  {
403  //Update Last valid heartbeat
404  mra_vc_last_hb[i] = last_hb_wid_timestamp;
405 
406  // Update Fail Timeout Period to worker
407  mra_vc_fail_timeout_period[i] = ((config_mra.failure_timeout_conf * config_mra.mra_heartbeat_interval) + control_timestamp );
408  break;
409  }
410  i++;
411  }
412 
413  for (i=0; i < config_mra.mra_number_of_workers; i++)
414  {
416  if( (control_timestamp - mra_vc_last_hb[i]) <= config_mra.mra_heartbeat_interval )
417  {
418  //worker active
419  vc_state_working[i] = 1;
420  }
421  else
422  { //worker inactive
423  vc_state_working[i] = 0;
424  }
426  if ((control_timestamp > mra_vc_fail_timeout_period[i] )
427  && (control_timestamp - mra_vc_last_hb[i]) > config_mra.mra_heartbeat_interval)
428  { // Failure state
429  mra_vc_state_failure[i] = 0;
430  }
431  else
432  { // Operation
433  mra_vc_state_failure[i] = 1;
434  }
435 
438  if (vc_state_working[i] == 0 && mra_vc_state_failure[i] == 0 )
439  {
440  mra_f_detec_f.mra_vc_status = VC_FAILURE;
441  mra_f_detec_f.mra_vc_wid = i;
442  behavior[mra_f_detec_f.mra_vc_wid] = mra_f_detec_f.mra_vc_status;
443  mra_f_detec_f.mra_last_hbtime = mra_vc_last_hb[i];
444 
445  // XBT_INFO ("%s, Failure detected %d \n", MSG_host_get_name(config_mra.workers_mra[i]), mra_vc_status);
446  }
447  else if (vc_state_working[i] == 0 && mra_vc_state_failure[i] == 1 )
448  {
449  mra_f_detec_f.mra_vc_status = VC_TRANSIENT;
450  mra_f_detec_f.mra_vc_wid = i;
451  behavior[mra_f_detec_f.mra_vc_wid] = mra_f_detec_f.mra_vc_status;
452  mra_f_detec_f.mra_last_hbtime = mra_vc_last_hb[i];
453  // XBT_INFO ("%s, Transient detected %d \n", MSG_host_get_name(config_mra.workers_mra[i]), mra_vc_status);
454  }
455  else if (vc_state_working[i] == 1 && mra_vc_state_failure[i] == 1 )
456  {
457  if (behavior[i] == VC_TRANSIENT)
458  {
459  mra_f_detec_f.mra_vc_status = VC_NORMAL;
460  mra_f_detec_f.mra_vc_wid = i;
461  behavior[mra_f_detec_f.mra_vc_wid] = mra_f_detec_f.mra_vc_status;
462  mra_f_detec_f.mra_last_hbtime = mra_vc_last_hb[i];
463  }
464  }
465  else
466  {
467  mra_f_detec_f.mra_vc_status = VC_UP_TRANSIENT;
468  mra_f_detec_f.mra_vc_wid = i;
469  }
470 
472  fprintf(failure_log,"Control_timestamp %g, %s, Last_heartb %g, Fta %g, Machine State %d, Failure State %d, Status %d, Trigger %zd \n",
473  control_timestamp, MSG_host_get_name(config_mra.workers_mra[i]), mra_vc_last_hb[i],
474  mra_vc_fail_timeout_period [i], vc_state_working[i], mra_vc_state_failure[i], mra_f_detec_f.mra_vc_status, mra_wid );
475  }
