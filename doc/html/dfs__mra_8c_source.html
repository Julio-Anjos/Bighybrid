466  {

468  {
469  chunk_owner_mra[chunk][i] = 1;
470  mra_affinity[chunk] = mra_affinity[chunk] + 1;

472  }
473  }
474  if(mra_affinity[chunk] == config_mra.mra_chunk_replicas)

476  fprintf (log,"mra_wid: %u \t chunk: %zu \t Replicas:%u \n", i, chunk, mra_affinity[chunk]);
477  }
478  }
479  }
480  }
481  /* @brief Creates a replica to owner in different groups */
482  for (owner = 0; owner < config_mra.mra_number_of_workers ; owner++)
483  {
484  for (chunk = 0; (chunk < config_mra.mra_chunk_count) ; chunk++)
485  {
486  if (chunk_owner_mra[chunk][owner] == 1)
487  {
488  while (mra_affinity[chunk] < config_mra.mra_chunk_replicas)
489  {
490  j = rand () % config_mra.mra_number_of_workers + 0;
491  while (owner == j)
492  {
493  j = rand () % mra_dist_manage.max_tot_dist + mra_dist_manage.min_tot_dist;
494  }
495  if (dist_bruta[owner] != dist_bruta[j] && chunk_owner_mra[chunk][j] == 0 )
496  {
497  if ((mra_affinity[chunk] < config_mra.mra_chunk_replicas) )
498  {
499  chunk_owner_mra[chunk][j] = 1;
500  mra_affinity[chunk] = mra_affinity[chunk] + 1;
501  }
502  }
503  }
504  if(mra_affinity[chunk] == config_mra.mra_chunk_replicas)
505  {
506  fprintf (log,"mra_wid: %u \t chunk: %zu \t Replicas:%u \n", i, chunk, mra_affinity[chunk]);
